% Shared admonition boxes for the book (LaTeX/PDF output).
% Beamer already provides block/alertblock/examples/definition via the theme,
% so this file is for the book only. The Lua filter brand-divs.lua maps
% fenced divs (::: {.note}, ::: {.warning}, etc.) to these environments.
%
% Requires: tcolorbox with breakable,skins; brand-colors.tex loaded first.

\newtcolorbox{note}[1][]{%
	enhanced, breakable,
	colback=brandAccentSoft, colframe=brandAccent,
	coltitle=white, fonttitle=\bfseries,
	title={#1}, arc=2mm,
	boxrule=0.6pt, left=3mm, right=3mm, top=2mm, bottom=2mm
}

\newtcolorbox{warning}[1][]{%
	enhanced, breakable,
	colback=red!5!white, colframe=red!70!black,
	coltitle=white, fonttitle=\bfseries,
	title={#1}, arc=2mm,
	boxrule=0.6pt, left=3mm, right=3mm, top=2mm, bottom=2mm
}
