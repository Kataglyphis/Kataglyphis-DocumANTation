\newglossaryentry{gpu}
{
    name=GPU,
    description={Graphics Processing Unit -- a specialized processor designed for parallel computation, primarily used for rendering graphics}
}

\newglossaryentry{cpu}
{
    name=CPU,
    description={Central Processing Unit -- the general-purpose processor that executes application logic}
}

\newglossaryentry{api}
{
    name=API,
    description={Application Programming Interface -- a set of protocols and tools for building software}
}

\newglossaryentry{brdf}
{
    name=BRDF,
    description={Bidirectional Reflectance Distribution Function -- describes how light is reflected at a surface point}
}

\newglossaryentry{pbr}
{
    name=PBR,
    description={Physically Based Rendering -- a rendering approach that models light transport using physically motivated equations}
}

\newglossaryentry{bvh}
{
    name=BVH,
    description={Bounding Volume Hierarchy -- a tree structure of bounding volumes used to accelerate ray-scene intersection tests}
}

\newglossaryentry{ndc}
{
    name=NDC,
    description={Normalized Device Coordinates -- the coordinate space after perspective divide, where all visible coordinates lie in $[-1, 1]^3$}
}

\newglossaryentry{fov}
{
    name=FOV,
    description={Field of View -- the angular extent of the observable world seen through the camera}
}

\newglossaryentry{msaa}
{
    name=MSAA,
    description={Multisample Anti-Aliasing -- a technique that reduces aliasing by sampling each pixel at multiple sub-pixel locations}
}

\newglossaryentry{taa}
{
    name=TAA,
    description={Temporal Anti-Aliasing -- an anti-aliasing technique that accumulates multiple frames with sub-pixel jitter}
}

\newglossaryentry{ssao}
{
    name=SSAO,
    description={Screen-Space Ambient Occlusion -- an approximation of ambient occlusion computed from the depth buffer}
}

\newglossaryentry{ssr}
{
    name=SSR,
    description={Screen-Space Reflections -- a technique that computes reflections by ray-marching through the depth buffer}
}

\newglossaryentry{gbuffer}
{
    name=G-Buffer,
    description={Geometry Buffer -- a set of screen-space textures storing per-pixel geometry data for deferred shading}
}

\newglossaryentry{uvmapping}
{
    name=UV mapping,
    description={The process of projecting a 2D texture onto a 3D surface using texture coordinates $(u, v)$}
}

\newglossaryentry{mipmap}
{
    name=Mipmap,
    description={A pre-computed sequence of textures at decreasing resolutions, used to reduce aliasing when textures are viewed at a distance}
}

\newglossaryentry{spirv}
{
    name=SPIR-V,
    description={Standard Portable Intermediate Representation -- a binary intermediate language for shaders, consumed by Vulkan}
}

\newglossaryentry{gbuffer-def}
{
    name=Deferred Shading,
    description={A rendering technique that separates geometry rasterization from lighting computation using a G-Buffer}
}

\newglossaryentry{ggx}
{
    name=GGX,
    description={Trowbridge-Reitz microfacet normal distribution function, the current standard for PBR specular}
}

\newglossaryentry{sah}
{
    name=SAH,
    description={Surface Area Heuristic -- a cost model used to optimize BVH construction by minimizing expected traversal cost}
}

\newglossaryentry{pcf}
{
    name=PCF,
    description={Percentage-Closer Filtering -- a shadow map filtering technique that softens shadow edges by averaging depth comparisons}
}

\newacronym{ldr}{LDR}{Low Dynamic Range}
\newacronym{ibl}{IBL}{Image-Based Lighting}

\newacronym{fps}{FPS}{Frames per Second}
\newacronym{lod}{LOD}{Level of Detail}
